library(babynames)
library(dplyr)
library(highcharter)

NAMES <- "Britney"
# c("Anna",  "Britney", "Whitney")

data <- babynames::babynames %>% 
  filter(name %in% NAMES) %>% 
  select(name, x = year, y = prop) %>% 
  group_by(name, x) %>% 
  summarise(y = sum(y), .groups = "drop") %>% 
  select(x, y)

data <- data %>% 
  mutate(
    # color = colorize(y, colors = viridisLite::inferno(100, begin = 0.2, end = 0.8)),
    color = colorize(x, colors = viridisLite::inferno(100, begin = 0.2, end = 0.8)),
    alpha = scales::rescale(sqrt(y), to = c(0.05, .75)),
    alpha = round(alpha, 2),
    color = purrr::map2_chr(color, alpha, shades::opacity)
  )

zns <- data %>% 
  select(value = x, color) %>% 
  distinct(value, .keep_all = TRUE) %>% 
  mutate(value = value + 0.5) %>% 
  list_parse()

data <- data %>% 
  select(x, y) %>% 
  mutate_all(round, 5)

highchart() %>%
  hc_add_series(
    data,
    "areaspline",
    hcaes(x, y),
    zones = zns,
    enableMouseTracking = FALSE,
    linkedTo = "serie"
  ) %>%
  hc_add_series(
    data,
    "spline",
    hcaes(x, y),
    color = "white",
    lineWidth = 8,
    hcaes(x, y),
    enableMouseTracking = FALSE,
    linkedTo = "serie"
  ) %>%
    hc_add_series(
    data,
    "spline",
    hcaes(x, y),
    zones = zns,
    lineWidth = 4,
    id = "serie"
  ) %>%
  hc_plotOptions(
    series = list(
      showInLegend = FALSE,
      zoneAxis = "x",
      marker = list(enabled = FALSE),
      hover = list(halo = list(size = 0))
    )
  ) %>% 
  hc_yAxis(visible = FALSE) %>% 
  hc_xAxis(
    lineColor = "transparent",
    tickLength = 0,
    tickInterval = 1,
    labels = list(
      style = list(color = "gray"),
      useHTML = TRUE,
      formatter = JS("function(){  var v = (this.value % 5 == 0) ? '<span>&#9679</span>' : '';  return v; }"))
    )
  ) %>% 
  hc_size(height = 300)
